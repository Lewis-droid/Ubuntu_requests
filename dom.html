Variables, conditionals, and user input

// Conditional example
function checkAge() {
  let age = document.getElementById('userAge').value;
  let output = "";
  if (age >= 18) {
    output = "You are eligible to vote!";
  } else if (age > 0) {
    output = `You will be eligible to vote in ${18 - age} years.`;
  } else {
    output = "Please enter a valid age.";
  }
  document.getElementById('ageOutput').textContent = output;
}

// Function example
function calculate(operation) {
  let num1 = parseFloat(document.getElementById('num1').value);
  let num2 = parseFloat(document.getElementById('num2').value);
  let result;
  if (operation === 'add') {
    result = num1 + num2;
  } else if (operation === 'multiply') {
    result = num1 * num2;
  }
  document.getElementById('calcOutput').textContent = `Result: ${result}`;
}

Part 3: JavaScript Loops

Repetition and iteration

Loop output will appear here
// Loop example
function generateMultiplicationTable() {
  let output = "Multiplication Table (1-5):\
";
  for (let i = 1; i <= 5; i++) {
    for (let j = 1; j <= 10; j++) {
      output += `${i} x ${j} = ${i * j}\
`;
    }
    output += "\
";
  }
  document.getElementById('loopOutput').innerHTML = output;
}

function changeColor() {
